clangd crashes when deducing template arguments

Source Code & Bug Trace:
#include <array>
#include <iostream>

template<typename T>
struct ArrayType {
    template<size_t size>
    using Array = std::array<T, size>;
};

template<ArrayType<int>::Array array>
void test() {
    for (const auto& item : array) {
        std::cout << item << std::endl;
    }
}

void foo() {
    test<{1,2,3}>();
}
