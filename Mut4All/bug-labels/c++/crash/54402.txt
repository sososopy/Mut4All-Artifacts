#include <utility>

template <typename T> int DefaultValue();

// Invoke DefaultValue() on all members of the tuple.
template <typename T, std::size_t... Is>
T DefaultValueTupleImpl(std::index_sequence<Is...>) {
  return std::tuple{[&]() {
    using Type = std::tuple_element_t<Is, T>;
    return DefaultValue<Type>();
  }()...};
}
