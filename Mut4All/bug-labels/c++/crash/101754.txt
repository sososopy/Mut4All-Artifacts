#include <concepts>

template <typename... Ts>
struct Overloaded : Ts... {
    using Ts::operator()...;
};

template <typename... Ts>
Overloaded(Ts...) -> Overloaded<Ts...>;

template <typename... Ts>
auto foo() {
    return Overloaded{
        [](std::same_as<Ts> auto) {} ...
    };
}

int main() {
    foo<int, double*>()(42);
    return 0;
}
