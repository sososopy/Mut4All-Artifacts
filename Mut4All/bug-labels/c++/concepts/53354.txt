Issue Title:
Segfault from constraining partial specialization on nested template type

Source Code & Bug Trace:
template <template <template <class> class> class>
struct void_t
{};

template <class T>
concept C1 = requires
{
  typename void_t<T::template value_types>;
};

template <class T>
concept C2 = requires (T& t)
{
  requires sizeof(t) > 4;
};

template <class T>
  requires C1<T>
struct A
{
};

template <class T>
  requires C1<T> && C2<T>
struct A<T>
{};

int main() {}
