#include <concepts>
#include <utility>


template <typename Type>
struct some_struct
{
    template <typename O>
    void test(Type& target, O&& value) &
        requires std::assignable_from<Type&, decltype(std::forward<O>(value))>;
};

template <typename Type>
template <typename O>
void some_struct<Type>::test(Type& target, O&& value) &
    requires std::assignable_from<Type&, decltype(std::forward<O>(value))>
{
    target = std::forward<O>(value);
}

int main()
{
    some_struct<int> a;

    int val = 10;
    a.test(val, 20);

    return 0;
}
