Issue Title:
ICE on unsound usage of existential types · Issue #52843 · rust-lang/rust · GitHub

Rust Code:
#![feature(existential_type)]

existential type Foo<T>: Default;

#[allow(unused)]
fn foo<T: Default>(t: T) -> Foo<T> {
    t
}

struct NotDefault;

fn main() {
    let _ = Foo::<NotDefault>::default();
}
