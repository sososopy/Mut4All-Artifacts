Issue Title:
ICE in conservative_impl_trait over VecDeque<String> Iterator · Issue #42399 · rust-lang/rust · GitHub

Rust Code:
#![feature(conservative_impl_trait)]

use std::collections::VecDeque;

struct Dude {
    pets: VecDeque<String>
}

impl Dude {
    pub fn new() -> Self {
        let mut pets = VecDeque::new();
        pets.push_front("Ardo".to_string());
        pets.push_front("Jamie".to_string());
        pets.push_front("Bowser".to_string());

        Dude {
            pets: pets
        }
    }

    pub fn iter_pets(&self) -> impl Iterator<Item=&str> {
        self.pets.iter()
    }
}

fn main() {
    let dude = Dude::new();
    for pet in dude.iter_pets() {
        println!("Pet: {}", pet);
    }
}
