Issue Title:
ICE using function pointer to function returning `impl Iterator<Item = &&'a str>` · Issue #50881 · rust-lang/rust · GitHub

Rust Code:
struct S<'a> {
    v: Vec<&'a str>
}

impl<'a> S<'a> {
    fn inner_iter(&self) -> impl Iterator<Item = &&'a str> {
        self.v.iter()
    }
}

fn main() {
    let s = S { v: Vec::new() };
    [s].iter()
        .flat_map(S::inner_iter)
        .for_each(|_| {});
}
