Issue Title:
Pattern-matching array values in struct fields causes compiler panic · Issue #57472 · rust-lang/rust · GitHub

Rust Code:
pub struct Punned {
    foo: [u8; 1],
    bar: [u8; 1],
}
pub fn test(punned: Punned) {
    match punned {
        Punned { foo: [_], .. } => println!("foo"),
        Punned { bar: [_], .. } => println!("bar"),
    }
}
