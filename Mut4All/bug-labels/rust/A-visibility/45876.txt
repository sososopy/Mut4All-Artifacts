ICE: Use a type that has already been pub used. · Issue #45876 · rust-lang/rust · GitHub

Rust Code:
mod visibility;
pub use self::visibility::*;

mod item_raw;
pub use self::item_raw::*;

use syntactic_analysis::nodes::Node;
use syntactic_analysis::nodes::Visibility;



pub struct Item {
    pub visibility: Option<Visibility>,
    pub item_raw: ItemRaw
}



impl Node for Item {
    
    fn get_nodes(&self) -> Vec<&Node> {
        match self.visibility {
            Some(ref visibility) => vec![
                visibility, 
                &self.item_raw
            ], 
            None => vec![
                &self.item_raw
            ]
        }
    }
}
