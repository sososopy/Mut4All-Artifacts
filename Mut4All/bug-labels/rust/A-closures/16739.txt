Issue Title:
ICE: When creating Box<FnOnce<(), ()>> from unsugared unboxed closure · Issue #16739 · rust-lang/rust · GitHub

Rust Code:
#![feature(unboxed_closures)]

struct Foo;

impl FnOnce<(), ()> for Foo {
    extern "rust-call" fn call_once(self, _: ()) {}
}

fn main() {
    // OK
    let _ = (box |:| {}) as Box<FnOnce<(), ()>>;

    // ICE
    let _ = box Foo as Box<FnOnce<(), ()>>;
}
