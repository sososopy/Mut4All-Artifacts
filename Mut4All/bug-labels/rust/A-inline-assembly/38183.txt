Issue Title:
LLVM assertion when trying to use fn items as inline assembly parameters · Issue #38183 · rust-lang/rust · GitHub

Rust Code:
#![feature(asm)]
fn abcd() {
    println!("lo");
}
fn main() {
    unsafe{asm!(".global abc;.set abc, ${0:P}" : : "i"(abcd as fn()))};
    println!("hi");
    unsafe { abc() };
}
extern {
    fn abc();
}
