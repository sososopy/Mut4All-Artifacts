Issue Title:
thread 'rustc' has overflowed its stack (when using #[derive(serde::Deserialize)] inside a custom Deserialize impl) · Issue #81844 · rust-lang/rust · GitHub

Rust Code:
struct AB {}

impl<'de> serde::Deserialize<'de> for AB {
    fn deserialize<D>(_deserializer: D) -> Result<Self, D::Error>
    where
        D: serde::Deserializer<'de>,
    {
        #[derive(serde::Deserialize)]  // comment this out to fix the stack overflow
        struct A2 {}
        
        return Ok(AB{})
    }
}

fn main(){}
