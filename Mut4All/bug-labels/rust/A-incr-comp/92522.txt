fn to_lines(path: &PathBuf) -> Vec<String> {                                                                                                                                                                                                                                                                                   
    let s = fs::read_to_string(path)                                                                                                                                                                                                                                                                                           
        .expect(format!("failed to read {:?}", &path).as_str());                                                                                                                                                                                                                                                               
                                                                                                                                                                                                                                                                                                                               
    s.par_lines()                                                                                                                                                                                                                                                                                                              
        .filter(|str| {                                                                                                                                                                                                                                                                                                        
            str.chars().nth(0) != Some('#')                                                                                                                                                                                                                                                                                    
        })                                                                                                                                                                                                                                                                                                                     
        .map(|str| str.to_string())                                                                                                                                                                                                                                                                                            
        .collect::<Vec<String>>()                                                                                                                                                                                                                                                                                              
}
