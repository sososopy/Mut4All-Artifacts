Issue Title:
ICE when using nested macro calls: 'no entry found for key' · Issue #44851 · rust-lang/rust · GitHub

Rust Code:
macro_rules! make_foodoc {
    ($outer:ident, $inner:ident) => {
        concat!("outer: ", stringify!($outer), " ; inner: ", stringify!($inner))
    }
}

macro_rules! make_foo {
    ($doc:expr, $outer:ident, $inner:ident) => {
        #[doc = $doc ]
        #[derive(Debug)]
        pub struct $outer($inner);
    }
}

make_foo!(make_foodoc!(MyFoo, i32), MyFoo, i32);

fn main() {
    let f = MyFoo(0);
    println!("{:?}", f);
}
