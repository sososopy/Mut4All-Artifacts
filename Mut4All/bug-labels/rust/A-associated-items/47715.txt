Issue Title:
Compiler crash when using impl Trait to bound associated type parameter · Issue #47715 · rust-lang/rust · GitHub

Rust Code:
#![feature(universal_impl_trait)]

trait Foo {}

trait Iterable {
    type Item;
    type Iter: Iterator<Item=Self::Item>;
    fn iterator(&self) -> Self::Iter;
}

struct Container<T: Iterable<Item = impl Foo>> {
    field: T
}
