Issue Title:
ICE when using const_fn in trait · Issue #54954 · rust-lang/rust · GitHub

Rust Code:
#![feature(const_fn)]

const ARR_LEN: usize = Tt::const_val::<[i8; 123]>();

trait Tt {
    const fn const_val<T: Sized>() -> usize {
        core::mem::size_of::<T>()
    }
}

fn f(z: [f32; ARR_LEN]) -> [f32; ARR_LEN] {
    z
}

fn main() {
    let _ = f([1f32; ARR_LEN]);
}
