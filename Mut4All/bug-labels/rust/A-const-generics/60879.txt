Issue Title:
ICE on const generics in impl block ("failed to lift"). · Issue #60879 · rust-lang/rust · GitHub

Rust Code:
#![feature(const_generics)]

struct Foo<T, const N: usize>([T; {N}]);

impl<T, const N: usize> Foo<T, {N}> {
    fn foo(&self) -> usize {
        {N}
    }
}

fn main() {
    let foo = Foo([0u32; 21]);
}
