use std::future::Future;

pub trait AsyncReadFrom: Sized {

    fn async_read_from<R: Unpin>(r: &mut R) -> Self::Output<R>;

    fn async_read_from<R: Unpin>(r: &mut R) -> Self::Output<R> {

        async move {
            while true {
                T::async_read_from(r.get_mut()).await;

        }
