Issue Title:
`Box<T, &A>` and LTO: Explicit gep type does not match pointee type of pointer operand · Issue #95036 · rust-lang/rust · GitHub

Rust Code:
#![feature(allocator_api)]
#![feature(bench_black_box)]
use std::alloc::Global;
use std::hint::black_box;

pub fn f() {
    let mut node = Box::new_in(Node {
        v: &(),
        b: false,
    }, &Global);
    node.b = true;
    black_box(node); // without this bug only triggers when built without optimizations
}

struct Node {
    v: &'static (),
    b: bool,
}
