Issue Title:
Compiler error when repeating GAT in FnMut signature instead of using Self::AssociatedType syntax · Issue #100689 · rust-lang/rust · GitHub

Rust Code:
#![feature(generic_associated_types)]

struct Foo<'a> {
    foo: &'a mut usize,
}

trait Bar<'a> {
    type FooRef<'b>
    where
        'a : 'b,
    ;
    fn uwu (
        foo: Foo<'a>,
        f: impl for<'b> FnMut(Self::FooRef<'b>),
    )
    ;
}
impl<'a> Bar<'a> for () {
    type FooRef<'b>
    =
        &'b Foo<'a>
    where
        'a : 'b,
    ;
    
    fn uwu (
        foo: Foo<'a>,
        mut f: impl for<'b> FnMut(&'b Foo<'a>), //relevant part
    )
    {
        f(&foo);
    }
}
