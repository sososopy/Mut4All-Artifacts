Issue Title:
Infinite loop in macro expansion · Issue #3829 · rust-lang/rust · GitHub

Rust Code:
fn macros() {
    macro_rules! if_ok(
        ($inp: expr) => (
            match $inp {
                Ok(move v) => { move v }
                Err(move e) => { return Err(e); }
            }
        )
    );
}

fn main() {
    let r = Ok(2);
    let p = if_ok!(move r);
    assert p == 2;
}
