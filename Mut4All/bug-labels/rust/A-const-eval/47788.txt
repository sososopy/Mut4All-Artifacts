ICE when combining const (fn) and unions · Issue #47788 · rust-lang/rust · GitHub

Rust Code:
#![feature(const_fn)]

use std::time::Instant;

pub struct GlobalTimer {
    time: Instant,
}

union DummyUnion {
    time: Instant,
    junk: u32
}

impl GlobalTimer {
    pub const fn init() -> GlobalTimer {
        const DUMMY: DummyUnion = DummyUnion {
            junk: 0
        };
        
        const UNINIT: Instant = unsafe {
            DUMMY.time
        };
        
        // Ta-da!
        GlobalTimer {
            time: UNINIT,
        }
    }
}
