// Ctor(..) is transformed to Ctor { 0: ... } in HAIR lowering, so directly
// calling constructors doesn't require them to be const.

    let f = Some;
    let g = Err;

    let x = f(5);
    let y = g(false);
    let z = h((x, y));
    z
}

};


fn main() {
