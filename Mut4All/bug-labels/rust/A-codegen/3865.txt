Issue Title:
explicit self and object types (trait types) are incompatible · Issue #3865 · rust-lang/rust · GitHub

Rust Code:
trait Foo {
    fn foo(&self) -> uint;
}

impl uint: Foo {
    fn foo(&self) -> uint { *self }
}

fn bar(x: @Foo, v: uint) {
    assert x.foo() == v;
}

fn main() {
    let x: @uint = @3;
    bar(x as @Foo, *x);
}
