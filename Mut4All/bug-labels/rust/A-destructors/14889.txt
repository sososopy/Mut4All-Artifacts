Issue Title:
ICE: lifetime substitution in struct fields when struct implements Drop · Issue #14889 · rust-lang/rust · GitHub

Rust Code:
#[deriving(Show)]
pub struct Context<'a> {
    field: Nested<'a>,
}

#[deriving(Show)]
pub struct Nested<'a>;

#[cfg(not(no_drop))]
impl<'a> Drop for Context<'a> {
    fn drop(&mut self) {
        println!("Dropping {}", *self);
    }
}

fn main() {
    let c = Context { field: Nested };
    println!("{}", c);
}
