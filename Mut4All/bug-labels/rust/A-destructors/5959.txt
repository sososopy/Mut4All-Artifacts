Issue Title:
ICE with compound assignment and Drop::finalize · Issue #5959 · rust-lang/rust · GitHub

Rust Code:
struct matrix;

impl Drop for matrix {
    fn finalize(&self) {}
}

impl Add<matrix, matrix> for matrix {
    fn add(&self, _rhs: &matrix) -> matrix{
        matrix
    }
}

fn main(){
    let mut m = matrix;
    m += matrix; // an ICE is occured
    //m = m + matrix; // not occured
    info!(m);
}
