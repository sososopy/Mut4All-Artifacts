Issue Title:
ICE when matching on enum variant constructors as fn pointers · Issue #48205 · rust-lang/rust · GitHub

Rust Code:
enum Variant<T> {
    A(T),
    B(T),
}

fn main() {
    const A: fn(usize) -> Variant<usize> = Variant::A;
    const B: fn(usize) -> Variant<usize> = Variant::B;

    let my_variant: fn(usize) -> Variant<usize> = Variant::A;

    match my_variant {
        A => {}
        B => {}
        _ => {}
    }
}
