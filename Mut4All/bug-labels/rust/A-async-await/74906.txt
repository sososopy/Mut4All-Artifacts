Issue Title:
Async/.await and const-generics causes "Broken MIR:" · Issue #74906 · rust-lang/rust · GitHub

Rust Code:
#![feature(const_generics)]
#![allow(incomplete_features)]

const SIZE: usize = 16;

struct Bar<const H: usize> {}

struct Foo<const H: usize> {}

impl<const H: usize> Foo<H> {
    async fn biz(_: &[[u8; SIZE]]) -> Vec<()> {
        vec![]
    }

    pub async fn baz(&self) -> Bar<H> {
        Self::biz(&vec![]).await;
        Bar {}
    }
}
