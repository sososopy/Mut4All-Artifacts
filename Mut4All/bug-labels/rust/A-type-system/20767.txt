Issue Title:
Coherence fails to find ambiguity with manual impl of Fn for fn · Issue #20767 · rust-lang/rust · GitHub

Rust Code:
#![feature(unboxed_closures)]

use std::ops::Fn;

struct Foo(u32);

impl Fn(u32) -> Foo for fn(u32) -> Foo {
    extern "rust-call" fn call(&self, args: (u32,)) -> Foo { let (u,) = args; self(u) }
}

fn main() {
    let x = Foo.call((42,));
    println!("{}", x);
}
