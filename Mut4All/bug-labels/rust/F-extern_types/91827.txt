Issue Title:
ICE: Fields cannot be extern types, unless they are at offset 0 · Issue #91827 · rust-lang/rust · GitHub

Rust Code:
#![feature(extern_types)]

extern "C" {
    type Content;
}

pub struct List {
    len: usize,
    _opaque: Content,
}

pub fn main() {
    let _len = match Option::<&List>::None {
        Some(a) => a.len,
        None => 0,
    };
}
